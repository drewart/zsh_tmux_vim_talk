# modal editing

i - insert

I - insert at start

a - append

A - append at end

r - replace

o - open below line

O - open edit above line

d - delete

x - delete char

# exit + io

: - vim prompt 

:w - write file

:w <name> - write to another file name

:wq - write and quit

:q - quit

:q! - quit if 

:help -- for help

## movements

h j k l

w - move 1 word

2w - move 2 words

e - end of word

b - back word

^ - start of line

$ - end of line

ctrl+d - page down

ctrl+a - page up

ctrl+E - scroll down

ctrl+Y - scroll up

M - center

fx - forward to first char x

% - go to matching brace

~ - change case

## trick edits vim-

# quoting words vim-surround

csw' - change surround word single quote

csw" - change surround word quote

ci'  - change in quote 'here'

## visual edit

V - visual line select

v - visual section select

Ctrl+v - block visual select mode


## search

/ - forward search

? - backward search

## search replace

:%s/find/replace/g
